119def Sample2DTracersFromField(delta_sampling, mask, N_realisations, n_tracers, seed=None, map_NSIDE=64):
120    '''
121    Creates a number of realisations of discrete tracers sampled from a given 2D angular overdensity field using Poisson sampling.
122
123    Parameters
124    ----------
125    delta_sampling : float ndarray
126        HEALPix overdensity field from which tracers are to be sampled.
127
128    mask : int ndarray
129        HEALPix mask defining the survey footprint. Pixels within the footprint should have value 1, while pixels outside the footprint should have value ``hp.UNSEEN``.
130
131    N_realisations : int
132        total number of realisations to be sampled.
133
134    n_tracers : int
135        total number of tracers to be sampled.
136
137    seed : int, optional
138        random seed for reproducibility, by default None.
139
140    map_NSIDE : int, optional
141        HEALPix nside of the output number counts map, by default 64.
142
143    Returns
144    -------
145    
146    tracer_pos_masked_ds_arr : float ndarray
147        array of shape (``N_realisations``, ``n_tracers``, 2) containing the sampled tracer positions in radians. The last dimension contains (Dec, RA) pairs.
148
149    map : float ndarray
150        HEALPix map of the number counts of the sampled tracers, summed over all realisations.
151    '''
152
153    #-----------------------------------------------------------------------------------------------
154
155    NSIDE = hp.npix2nside(len(delta_sampling))
156    Nobj = int(1.5*n_tracers*12*NSIDE**2/len(np.where(mask==1)[0]))
157
158    tracer_pos_arr = np.zeros((N_realisations, Nobj, 2))
159    tracer_pos_masked_ds_arr = np.zeros((N_realisations, n_tracers, 2))
160
161    np.random.seed(seed)
162
163    for i, tracer_pos in enumerate(tracer_pos_arr):
164
165        sampled_pix = hpt.rand_pix_from_map(1+delta_sampling, n=Nobj)
166        sampled_RAs, sampled_Decs = hp.pix2ang(NSIDE, sampled_pix, lonlat=True)
167        tracer_pos[:, 0] = np.deg2rad(sampled_Decs)
168        tracer_pos[:, 1] = np.deg2rad(sampled_RAs)
169        ipix = hp.ang2pix(NSIDE, np.rad2deg(tracer_pos[:, 1]), np.rad2deg(tracer_pos[:, 0]), nest=False, lonlat=True)
170        mask_val = mask[ipix]
171        ind_masked = np.where(mask_val!=hp.UNSEEN)[0]
172        tracer_pos_masked = tracer_pos[ind_masked]
173        # print(len(tracer_pos_masked), n_tracers)
174        tracer_pos_masked_ds_arr[i] = tracer_pos_masked[np.random.choice(len(tracer_pos_masked), n_tracers, replace=False)]
175
176    map = cat2hpx(np.rad2deg(tracer_pos_masked_ds_arr)[:, :, 1], np.rad2deg(tracer_pos_masked_ds_arr)[:, :, 0], nside=map_NSIDE, radec=False).astype(float)
177
178    return tracer_pos_masked_ds_arr, map

182def Sample2DPoissonTracers(mask, N_realisations, n_tracers, seed=None, map_NSIDE=64):
183    '''
184    Creates a number of realisations of discrete tracers sampled from a uniform Poisson distribution over the sky.
185
186    Parameters
187    ----------
188    mask : int ndarray
189        HEALPix mask defining the survey footprint. Pixels within the footprint should have value 1, while pixels outside the footprint should have value ``hp.UNSEEN``.
190
191    N_realisations : int
192        total number of realisations to be sampled.
193
194    n_tracers : int
195        total number of tracers to be sampled.
196
197    seed : int, optional
198        random seed for reproducibility, by default None.
199
200    map_NSIDE : int, optional
201        HEALPix nside of the output number counts map, by default 64.
202
203    Returns
204    -------
205    
206    randoms_pos_masked_ds_arr : float ndarray
207        array of shape (``N_realisations``, ``n_tracers``, 2) containing the sampled tracer positions in radians. The last dimension contains (Dec, RA) pairs.
208
209    map : float ndarray
210        HEALPix map of the number counts of the sampled tracers, summed over all realisations.
211    '''
212
213    #-----------------------------------------------------------------------------------------------
214
215    NSIDE = hp.npix2nside(len(mask))
216    Nobj = int(1.5*n_tracers*12*NSIDE**2/len(np.where(mask==1)[0]))
217
218    randoms_pos_arr = np.zeros((N_realisations, Nobj, 2))
219    randoms_pos_masked_ds_arr = np.zeros((N_realisations, n_tracers, 2))
220
221    np.random.seed(seed)
222
223    for i, randoms_pos in enumerate(randoms_pos_arr):
224
225        randoms_pos[:, 0] = np.arcsin(np.random.uniform(low=-1, high=1, size=randoms_pos.shape[0]))
226        randoms_pos[:, 1] = np.random.uniform(low=0, high=2*np.pi, size=randoms_pos.shape[0])
227        ipix = hp.ang2pix(NSIDE, np.rad2deg(randoms_pos[:, 1]), np.rad2deg(randoms_pos[:, 0]), nest=False, lonlat=True)
228        mask_val = mask[ipix]
229        ind_masked = np.where(mask_val!=hp.UNSEEN)[0]
230        randoms_pos_masked = randoms_pos[ind_masked]
231        randoms_pos_masked_ds_arr[i] = randoms_pos_masked[np.random.choice(len(randoms_pos_masked), n_tracers, replace=False)]
232
233    map = cat2hpx(np.rad2deg(randoms_pos_masked_ds_arr)[:, :, 1], np.rad2deg(randoms_pos_masked_ds_arr)[:, :, 0], nside=map_NSIDE, radec=False).astype(float)
234
235    return randoms_pos_masked_ds_arr, map

239def Sample3DPoissonTracers(N_realisations, n_tracers, boxsize=1000, starting_seed=42):
240    '''
241    Creates a number of realisations of discrete tracers sampled from a uniform Poisson distribution in 3D space.
242
243    Parameters
244    ----------
245    N_realisations : int
246        total number of realisations to be sampled.
247
248    n_tracers : int
249        total number of tracers to be sampled.
250
251    boxsize : float, optional
252        size of the cubic box in which the tracers are to be sampled, by default 1000.0 Mpc/h.
253
254    starting_seed : int, optional
255        random seed for reproducibility, by default 42. This is the seed set for the first realisation,
256        Henceforth, the next ith realisation has seed = starting_seed + i.
257
258    Returns
259    -------
260    
261    randoms_pos_ds_arr : float ndarray
262        array of shape (``N_realisations``, ``n_tracers``, 3) containing the sampled tracer positions in Mpc/h.
263    '''
264
265    #-----------------------------------------------------------------------------------------------
266
267    randoms_pos_ds_arr = []
268
269    np.random.seed(starting_seed)
270
271    for i in range(N_realisations):
272
273        randoms_pos_ds_arr.append(np.random.uniform(low=0, high=boxsize, size=(n_tracers, 3)))
274    
275    randoms_pos_ds_arr = np.array(randoms_pos_ds_arr)
276
277    return randoms_pos_ds_arr

281def Sample2DTracersFromQuijoteBox(sim_num, tracer_type, mask, N_realisations, n_tracers, seed=None, map_NSIDE=64, DataPath='../kNNpy/Data'):
282    '''
283    Creates a number of realisations of discrete tracers sampled from dark-matter halo catalogues created using the Quijote simulation hight-resolution boxes[^1]. The 3D tracer positions are converted to angular 2D coordinates by projecting them onto the sky of a hypothetical observer at the centre of the simulation box. Currently supports two types of tracers, namely 'Galaxies' and 'Clusters' delineated based on a simple mass cut; galaxies are halos with mass ``M_halo >= 3e13 Msun/h and M_halo <= 1e14 Msun/h``, while clusters are halos with mass ``M_halo >= 1e14 Msun/h and M_halo <= 2.5e14 Msun/h``.
284
285    Parameters
286    ----------
287    sim_num : int
288        Quijote simulation realisation number to be used for sampling the tracers.
289
290    tracer_type : str
291        type of tracers to be sampled. Currently supports 'Galaxies' and 'Clusters'.
292
293    mask : int ndarray
294        HEALPix mask defining the survey footprint. Pixels within the footprint should have value 1, while pixels outside the footprint should have value ``hp.UNSEEN``.
295
296    N_realisations : int
297        total number of realisations to be sampled.
298
299    n_tracers : int
300        total number of tracers to be sampled. Must be less than the total number of tracers available in the given simulation box after projecting onto the virtual sky and applying the survey mask. If the observational footprint covers the entire sky, the maximum number of 'Clusters' ('Galaxies') that can be sampled is <enter limit here> (<enter limit here>). For partial sky coverage, this number might be significantly lower.
301
302    seed : int, optional
303        random seed for reproducibility, by default None.
304
305    map_NSIDE : int, optional
306        HEALPix nside of the output number counts map, by default 64.
307
308    DataPath : str, optional
309        path to the kNNpy Data directory, by default '../kNNpy/Data/'.
310
311    Returns
312    -------
313    
314    tracer_pos_masked_ds_arr : float ndarray
315        array of shape (``N_realisations``, ``n_tracers``, 2) containing the sampled tracer positions in radians. The last dimension contains (Dec, RA) pairs.
316
317    map : float ndarray
318        HEALPix map of the number counts of the sampled tracers, summed over all realisations.
319
320    References
321    ----------
322    [^1]: <add reference here>
323    '''
324
325    #-----------------------------------------------------------------------------------------------
326
327    #Defining the quijote box size
328    boxsize=1000.0  #Mpc/h
329
330    #Converting the simulation number into a simulation name string. For example, when sim_num is 5, name will be 'sim_005' and when sim_num is 50, name will be 'sim_050'
331    num = str(sim_num).zfill(3)
332
333    #Getting the NSIDE from the mask
334    NSIDE = hp.npix2nside(len(mask))
335    Nobj = int(1.5*n_tracers*12*NSIDE**2/len(np.where(mask==1)[0]))
336
337    if tracer_type=='Galaxies': 
338        
339        #Reading in the halo catalogs corresponding to the given simulation number
340        folder_ga = f'{DataPath}/Quijote_simulations/Fiducial_HR_rockstar/Galaxy_cat'
341        filepath_ga = '{a}/sim_{b}'.format(a = folder_ga, b = num)
342        ga_data_selected = pd.read_csv(filepath_ga)
343
344        #Positions of halos
345        all_ga_pos = np.zeros((len(ga_data_selected), 3), dtype = np.float32)
346        all_ga_pos[:, 0] = ga_data_selected.loc[:, '8'].values
347        all_ga_pos[:, 1] = ga_data_selected.loc[:, '9'].values
348        all_ga_pos[:, 2] = ga_data_selected.loc[:, '10'].values
349
350        #Converting 3D positions to sky angles, observer at centre of box
351        all_angpos = cartesian_corner_to_angles_centre(all_ga_pos[:, 0], all_ga_pos[:, 1], all_ga_pos[:, 2], boxsize, boxsize/2)
352
353
354    elif tracer_type=='Clusters':
355        
356        #Reading in the halo catalogs corresponding to the given simulation number
357        folder_cl = f'{DataPath}/Quijote_simulations/Fiducial_HR_rockstar/Cluster_cat'
358        filepath_cl = '{a}/sim_{b}'.format(a = folder_cl, b = num)
359        cl_data_selected = pd.read_csv(filepath_cl)
360
361        #Positions of halos
362        all_cl_pos = np.zeros((len(cl_data_selected), 3), dtype = np.float32)
363        all_cl_pos[:, 0] = cl_data_selected.loc[:, '8'].values
364        all_cl_pos[:, 1] = cl_data_selected.loc[:, '9'].values
365        all_cl_pos[:, 2] = cl_data_selected.loc[:, '10'].values
366
367        #Converting 3D positions to sky angles, observer at centre of box
368        all_angpos = cartesian_corner_to_angles_centre(all_cl_pos[:, 0], all_cl_pos[:, 1], all_cl_pos[:, 2], boxsize, boxsize/2)
369
370    else:
371        raise ValueError("Invalid tracer type specified. Supported types are 'Galaxies' and 'Clusters'.")
372    
373    tracer_pos_arr = np.zeros((N_realisations, Nobj, 2))
374    tracer_pos_masked_ds_arr = np.zeros((N_realisations, n_tracers, 2))
375
376    np.random.seed(seed)
377
378    for i, tracer_pos in enumerate(tracer_pos_arr):
379
380        sel_ind = np.random.choice(len(all_angpos), Nobj, replace=False)
381        tracer_pos = all_angpos[sel_ind]
382        ipix = hp.ang2pix(NSIDE, np.rad2deg(tracer_pos[:, 1]), np.rad2deg(tracer_pos[:, 0]), nest=False, lonlat=True)
383        mask_val = mask[ipix]
384        ind_masked = np.where(mask_val!=hp.UNSEEN)[0]
385        tracer_pos_masked = tracer_pos[ind_masked]
386        tracer_pos_masked_ds_arr[i] = tracer_pos_masked[np.random.choice(len(tracer_pos_masked), n_tracers, replace=False)]
387
388    map = cat2hpx(np.rad2deg(tracer_pos_masked_ds_arr)[:, :, 1], np.rad2deg(tracer_pos_masked_ds_arr)[:, :, 0], nside=map_NSIDE, radec=False).astype(float)
389
390    return tracer_pos_masked_ds_arr, map

394def Sample3DTracersFromQuijoteBox(tracer_type, N_realisations, n_tracers, ptype=[1],DataPath='../kNNpy/Data', starting_seed=42):
395    '''
396    For now it only supports one tracer_type: 'particles'. Other types like 'halos', can be included if and when necessry.

414def make_overdensity_3D(N_realisations, grid, ptype, do_RSD=False, MAS='CIC', axis=0, verbose=False, DataPath='../kNNpy/Data'):
415    overdensity_list=[]
416    for i in range(N_realisations):
417        snapshot = f'{DataPath}/Quijote_simulations/fiducial_LR/{i}/snapdir_004/snap_004'
418        delta = MASL.density_field_gadget(snapshot, ptype, grid, MAS, do_RSD, axis, verbose)
419        delta /= np.mean(delta, dtype=np.float64);  delta -= 1.0
420        overdensity_list.append(delta)
421    overdensity_list=np.array(overdensity_list)
422    return overdensity_list
